RCA Table Creation:

-- 1. Enable pgvector (already enabled if tickets table works)
CREATE EXTENSION IF NOT EXISTS vector;


-- ============================================================
-- 2. RCA KNOWLEDGE BASE TABLE
-- ============================================================
CREATE TABLE IF NOT EXISTS rca_knowledge_base (
    id                  BIGSERIAL PRIMARY KEY,

    -- Incident classification
    title               TEXT NOT NULL,
    category            TEXT NOT NULL,   -- Database, Network, Application, etc.


    -- What triggers this RCA entry
    keywords            TEXT NOT NULL,
    symptoms            TEXT NOT NULL,

    -- The actual RCA content
    root_cause          TEXT NOT NULL,
    affected_component  TEXT NOT NULL,
    resolution_steps    JSONB NOT NULL DEFAULT '[]',
    confidence          TEXT NOT NULL CHECK (confidence IN ('HIGH','MEDIUM','LOW')),

    -- metadata

    environment         TEXT DEFAULT 'Production',
    ci_asset            TEXT,
    recurrence_count    INT  DEFAULT 1,

    status              TEXT DEFAULT 'Active'
        CHECK (status IN ('Active','Deprecated','Under Review')),

    -- Semantic search embedding (384-dim, same model as tickets)
    embedding           VECTOR(384),

    last_used_at        TIMESTAMP,
    created_at          TIMESTAMP DEFAULT NOW(),
    updated_at          TIMESTAMP DEFAULT NOW()
);


-- ============================================================
-- 3. VECTOR INDEX
-- ============================================================
CREATE INDEX IF NOT EXISTS idx_rca_kb_embedding
ON rca_knowledge_base
USING ivfflat (embedding vector_cosine_ops)
WITH (lists = 100);


-- ============================================================
-- 4. VECTOR SEARCH FUNCTION
-- ============================================================
CREATE OR REPLACE FUNCTION match_rca_knowledge_base(
    query_embedding VECTOR(384),
    match_count     INT
)
RETURNS TABLE (
    id                  BIGINT,
    title               TEXT,
    category            TEXT,
    symptoms            TEXT,
    root_cause          TEXT,
    affected_component  TEXT,
    resolution_steps    JSONB,
    confidence          TEXT,
    ci_asset            TEXT,
    recurrence_count    INT,
    similarity          FLOAT
)
LANGUAGE sql STABLE
AS $$
    SELECT
        id, title, category, symptoms,
        root_cause, affected_component, resolution_steps,
        confidence, ci_asset, recurrence_count,
        1 - (embedding <=> query_embedding) AS similarity
    FROM rca_knowledge_base
    WHERE status = 'Active'
      AND embedding IS NOT NULL
    ORDER BY embedding <=> query_embedding
    LIMIT match_count;
$$;
